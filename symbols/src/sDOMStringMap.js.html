1 /**
  2  * @constructor
  3  * @param {Object} items Map of data. Should be keys to strings. All strings
  4  *   should be in camelCase notation.
  5  * @returns {sDOMStringMap} The mapped object.
  6  */
  7 var sDOMStringMap = function (items) {
  8   for (var key in items) {
  9     if (items.hasOwnProperty(key)) {
 10       this[key] = items[key];
 11     }
 12   }
 13   return this;
 14 };
