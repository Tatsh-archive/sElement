1 /**
  2  * Class list to emulate DOMSettableTokenList returned by classList for
  3  *   elements.
  4  * @augments sDOMTokenList
  5  * @constructor
  6  * @see http://www.whatwg.org/specs/web-apps/current-work/#domtokenlist
  7  * @param {Element} element DOM element.
  8  * @returns {sDOMSettableTokenList|DOMTokenList} The native browser
  9  *   DOMTokenList or sDOMSettableTokenList.
 10  */
 11 var sDOMSettableTokenList = function (element) {
 12   this.parent.constructor.call(this, element);
 13   if (this._isNative) {
 14     return element.classList;
 15   }
 16   /**
 17    * Regenerates the className attribute.
 18    * @private
 19    */
 20   this._regenerateString = function () {
 21     this._DOMElement.className = this._classes.join(' ');
 22   };
 23 
 24   return this;
 25 };
 26 /**
 27  * @type sDOMTokenList
 28  * @private
 29  */
 30 sDOMSettableTokenList.prototype = new sDOMTokenList();
 31 // TODO Fix warning
 32 // /**
 33 //  * Access to the parent class.
 34 //  * @private
 35 //  * @type sDOMTokenList
 36 //  */
 37 sDOMSettableTokenList.prototype.parent = sDOMTokenList.prototype;
 38 /**
 39  * Constructor.
 40  * @private
 41  * @type function(new:sDOMSettableTokenList,(Element|null))
 42  */
 43 sDOMSettableTokenList.prototype.constructor = sDOMSettableTokenList;
 44 /**
 45  * Add a token to the current string.
 46  * @param {string} token Token to add.
 47  */
 48 sDOMSettableTokenList.prototype.add = function (token) {
 49   this._classes.push(token);
 50   this.length = this._classes.length;
 51   this._regenerateString();
 52 };
 53 /**
 54  * Removes a token from the current string.
 55  * @param {string} token Token to remove.
 56  */
 57 sDOMSettableTokenList.prototype.remove = function (token) {
 58   var original = this._classes;
 59   this._classes = [];
 60 
 61   for (var i = 0; i < original.length; i++) {
 62     if (original[i] !== token) {
 63       this._classes.push(original[i]);
 64     }
 65   }
 66 
 67   this._regenerateString();
 68   this.length = this._classes.length;
 69 };
 70 /**
 71  * Toggle a token from the current string.
 72  * @param {string} token Token to toggle.
 73  * @returns {boolean} If the token exists, returns false. If the token does
 74  *   not exist and it is added, returns true.
 75  */
 76 sDOMSettableTokenList.prototype.toggle = function (token) {
 77   if (this.contains(token)) {
 78     this.remove(token);
 79     return false;
 80   }
 81 
 82   this.add(token);
 83   return true;
 84 };
