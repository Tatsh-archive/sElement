1 /**
  2  * Test browser data. This is generally not recommended for everything, and is
  3  *   intended to be a last resort when the version of the browser to target is
  4  *   known and the environment is controlled (such as when the user has
  5  *   absolutely no way to change the user agent).
  6  * @constructor
  7  */
  8 var sBrowser = function () {};
  9 /**
 10  * If the browser is IE.
 11  * @type (boolean|null)
 12  * @private
 13  */
 14 sBrowser.isIEStatic = null;
 15 /**
 16  * Browser conditional comment queries mapped to boolean values.
 17  * @type Object
 18  * @private
 19  */
 20 sBrowser.isIEVersionStatics = {};
 21 /**
 22  * Test if the browser is Internet Explorer using conditional comments.
 23  * @returns {boolean} <em>true</em> if the browser is Internet Explorer,
 24  *   <em>false</em> false otherwise.
 25  */
 26 sBrowser.isIE = function () {
 27   if (sBrowser.isIEStatic === null) {
 28     var div = document.createElement('div');
 29     div.style.display = 'none';
 30     div.innerHTML = "<!--[if IE]><div id='ie-find'></div><[endif]-->";
 31     document.body.appendChild(div);
 32 
 33     if (document.getElementById('ie-find')) {
 34       sBrowser.isIEStatic = true;
 35     }
 36     else {
 37       sBrowser.isIEStatic = false;
 38     }
 39 
 40     document.body.removeChild(div);
 41   }
 42 
 43   return sBrowser.isIEStatic;
 44 };
 45 /**
 46  * Test if the browser matches an Internet Explorer conditional comment string.
 47  *   This is the part after the if keyword. Example: <code>lt IE 9</code>.
 48  * @param {string} version Version or query string.
 49  * @returns {boolean} If the browser matches.
 50  */
 51 sBrowser.isIEVersion = function (version) {
 52   if (sBrowser.isIEVersionStatics[version] === undefined) {
 53     var div = document.createElement('div');
 54     div.style.display = 'none';
 55     div.innerHTML = "<!--[if IE " + version + "]><div id='ie-find'></div><[endif]-->";
 56     document.body.appendChild(div);
 57 
 58     if (document.getElementById('ie-find')) {
 59       sBrowser.isIEVersionStatics[version] = true;
 60     }
 61     else {
 62       sBrowser.isIEVersionStatics[version] = false;
 63     }
 64 
 65     document.body.removeChild(div);
 66   }
 67 
 68   return sBrowser.isIEVersionStatics[version];
 69 };
 70 /**
 71  * @type (boolean|null)
 72  * @private
 73  */
 74 sBrowser._canUploadFiles = null;
 75 /**
 76  * Checks if the browser can upload files.
 77  * @returns {boolean} If the browser can upload files.
 78  */
 79 sBrowser.canUploadFiles = function () {
 80   if (sBrowser._canUploadFiles === null) {
 81     var a = document.createElement('input'), ret;
 82     try {
 83       a.setAttribute('type', 'file');
 84       document.body.appendChild(a);
 85       ret = !a.disabled;
 86       document.body.removeChild(a);
 87       sBrowser._canUploadFiles = ret;
 88     }
 89     catch (e) {}
 90     sBrowser._canUploadFiles = true;
 91   }
 92 
 93   return sBrowser._canUploadFiles;
 94 };
