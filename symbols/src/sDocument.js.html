  8 /**
  9  * Adds an event.
 10  * @param {string} eventName Event name.
 11  * @param {function((sEvent|Event))} func Callback.
 12  * @param {boolean} [useCapture=false] Indicates whether or not the user wishes
 13  *   to initiate capture.
 14  * @returns {sDocument} The sDocument object to allow method chaining.
 15  */
 16 sDocument.prototype.addEventListener = function (eventName, func, useCapture) {
 17   if (useCapture === undefined) {
 18     useCapture = false;
 19   }
 20 
 21   var ieEventName = 'on' + eventName;
 22 
 23   if (sBrowser.isIEVersion('lt 9') && eventName === 'DOMContentLoaded') {
 24     ieEventName = 'onload';
 25   }
 26 
 27   if (document.addEventListener) {
 28     document.addEventListener(eventName, func, useCapture);
 29   }
 30   else if (window.attachEvent) {
 31     // TODO This should be handled by sEvent internally with an ID number so it can be detached
 32     window.attachEvent(ieEventName, function () {
 33       func.call(document, new sEvent(window.event));
 34     });
 35   }
 36 
 37   return this;
 38 };

 69 /**
 70  * Gets elements by class name. Based on work by Robert Nyman.
 71  * @param {string} className Class name.
 72  * @returns {NodeList|Array} Element node list.
 73  * @see http://code.google.com/p/getelementsbyclassname/
 74  */
 75 sDocument.prototype.byClassName = function (className) {
 76   var elm = document;
 77   var ret, doc = document, classesToCheck, classes = className.split(' '), j;
 78   var elements;
 79 
 80   if (elm.getElementsByClassName) {
 81     ret = elm.getElementsByClassName(className);
 82   }
 83   else if (doc.evaluate) {
 84     // do XPath query and return array
 85     var xhtmlNamespace = 'http://www.w3.org/1999/xhtml';
 86     var namespaceResolver = doc.documentElement.namespaceURI === xhtmlNamespace ? xhtmlNamespace : null;
 87     var node;
 88 
 89     ret = [];
 90     classesToCheck = '';
 91 
 92     for (j = 0; j < classes.length; j++) {
 93       classesToCheck += '[contains(concat(" ", @class, " "), " ' + classes[j] + ' ")]';
 94     }
 95 
 96     try {
 97       elements = doc.evaluate('.//*' + classesToCheck, elm, namespaceResolver, 0, null);
 98     }
 99     catch (e) {
100       elements = doc.evaluate('.//*' + classesToCheck, elm, null, 0, null);
101     }
102 
103     while ((node = elements.iterateNext())) {
104       ret.push(node);
105     }
106   }
107   else {
108     var match, k;
109 
110     ret = [];
111     elements = elm.all ? elm.all : elm.getElementsByTagName('*');
112     classesToCheck = [];
113 
114     for (j = 0; j < classes.length; j++) {
115       classesToCheck.push(new RegExp('(\\s+)?' + classes[j] + '(\\s+)?'));
116     }
117 
118     for (j = 0; j < elements.length; j++) {
119       match = false;
120       for (k = 0; k < classesToCheck.length; k++) {
121         if (classesToCheck[k].test(elements[j].className)) {
122           match = true;
123           break;
124         }
125       }
126       if (match) {
127         ret.push(elements[j]);
128       }
129     }
130   }
131 
132   return ret;
133 };

142 /**
143  * Query selector. Uses native querySelectorAll() if available and has a
144  *   fallback for old browsers.<br>
145  * Note that often :visited and :link pseudo selectors will be ignored. This is
146  *   often for privacy reasons.<br>
147  * For any browser, this only supports what that browser supports. If the
148  *   browser lacks CSS 3 selectors, then this will return an empty NodeList
149  *   or Array.
150  * @returns {NodeList|Array} Iterable set of elements.
151  */
152 sDocument.prototype.querySelectorAll = function (selector) {
153   var doc = document;
154 
155   if (doc.querySelectorAll) {
156     return doc.querySelectorAll(selector);
157   }
158 
159   // IE 7 mainly
160   // Based on code by Chad Scira (unobfuscated here)
161   // https://gist.github.com/868532/778c45c40186a862642f72dd5b532abbe18a3625
162   // (function(d){d=document,a=d.styleSheets[0]||d.createStyleSheet();d.querySelectorAll=function(e){a.addRule(e,'f:b');for(var l=d.all,b=0,c=[],f=l.length;b<f;b++)l[b].currentStyle.f&&c.push(l[b]);a.removeRule(0);return c}})()
163   // Has a bug though, it should be to remove the last rule not the first (0)
164 
165   var styleSheet = doc.styleSheets[0] || doc.createStyleSheet();
166   var all = doc.all, elements = [];
167 
168   return (function (s) {
169     styleSheet.addRule(s, 'f:b');
170 
171     if (!all[0].currentStyle) { // If the browser lacks currentStyle
172       return [];
173     }
174 
175     for (var i = 0; i < all.length; i++) {
176 
177       if (all[i].currentStyle['f']) {
178         elements.push(all[i]);
179       }
180     }
181 
182     styleSheet.removeRule(styleSheet.rules.length - 1);
183 
184     return elements;
185   })(selector);
186 };
187 /**
188  * Query selector. Uses native querySelector() if available and has a
189  *   fallback for older browsers.<br>
190  * Note that in :visited and :link pseudo selectors will be ignored. This is
191  *   often for privacy.
192  * @returns {Element|null} DOM element or null.
193  */
194 sDocument.prototype.querySelector = function (selector) {
195   var doc = document;
196 
197   if (doc.querySelector) {
198     return doc.querySelector(selector);
199   }
200 
201   var element = this.querySelectorAll(selector)[0];
202 
203   if (element === undefined) {
204     return null;
205   }
206 
207   return element;
208 };
