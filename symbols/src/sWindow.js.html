1 /**
  2  * Front-end to window object properties such as height.
  3  * @constructor
  4  */
  5 var sWindow = function () {};
  6 /**
  7   * Add an event listener to the window object.
  8   * @param {string} type Type of event.
  9   * @param {(EventListener|function((sEvent|Event|null)):(boolean|undefined)|null)} func Callback.
 10   * @param {boolean} [useCapture=false] If the user is to initiate the event.
 11   * @returns {sWindow} The object to allow method chaining.
 12   */
 13 sWindow.prototype.addEventListener = function (type, func, useCapture) {
 14   if (window.addEventListener) {
 15     window.addEventListener(type, func, !!useCapture);
 16   }
 17   else if (window.attachEvent) {
 18     window.attachEvent('on' + type, function () {
 19       var event = new sEvent(window.event);
 20       var ret = func.call(sWindow, event);
 21       if (!ret) {
 22         event.preventDefault();
 23       }
 24       return ret;
 25     });
 26   }
 27   return this;
 28 };
 29 /**
 30  * Used in getHeight/width calculation.
 31  * @private
 32  * @type Element
 33  */
 34 sWindow._doc = (function (doc) {
 35   var ret = document.body;
 36   if (doc.compatMode === 'CSS1Compat' && doc.documentElement) {
 37     ret = doc.documentElement;
 38   }
 39   return ret;
 40 })(document);
 41 /**
 42  * Gets the height of the window.
 43  * @returns {number} Height of the window.
 44  */
 45 sWindow.prototype.getHeight = function () {
 46   var height = 460;
 47 
 48   sWindow._doc.offsetHeight && (height = sWindow._doc.offsetHeight);
 49   window.innerHeight && (height = window.innerHeight);
 50 
 51   return height;
 52 };
 53 /**
 54  * Gets the width of the window.
 55  * @returns {number} Width of the window.
 56  */
 57 sWindow.prototype.getWidth = function () {
 58   var width = 630;
 59   var doc = document.body;
 60 
 61   sWindow._doc.offsetWidth && (width = sWindow._doc.offsetWidth);
 62   window.innerWidth && (width = window.innerWidth);
 63 
 64   return width;
 65 };
 66 /**
 67  * Alias to addEventListener for window.
 68  * @param {string} type Type of event.
 69  * @param {(EventListener|function((sEvent|Event|null)):(boolean|undefined)|null)} func Callback.
 70  * @param {boolean} [useCapture] Not used.
 71  * @returns {sWindow} The object to allow method chaining.
 72  */
 73 sWindow.prototype.bind = function (type, func, useCapture) {
 74   return this.addEventListener(type, func, useCapture);
 75 };
 76 /**
 77  * Alias to addEventListener for window.
 78  * @param {string} type Type of event.
 79  * @param {(EventListener|function((sEvent|Event|null)):(boolean|undefined)|null)} func Callback.
 80  * @param {boolean} [useCapture] Not used.
 81  * @returns {sWindow} The object to allow method chaining.
 82  */
 83 sWindow.prototype.addEvent = function (type, func, useCapture) {
 84   return this.addEventListener(type, func, useCapture);
 85 };
