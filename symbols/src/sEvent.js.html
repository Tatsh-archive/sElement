1 /**
  2  * Emulates the event object.
  3  * @see https://developer.mozilla.org/en/DOM/DOM_event_reference
  4  * @constructor
  5  * @param {Event|Object} eventObj Native browser event or object that is
  6  *   similar (this includes IE's window.event).
  7  * @returns {sEvent} The event object.
  8  */
  9 var sEvent = function (eventObj) {
 10   /**
 11    * If the event is an IE event.
 12    * @type boolean
 13    * @private
 14    */
 15   this._isIE = window.event && window.event.srcElement !== undefined ? true : false;
 16   /**
 17    * The native browser event object.
 18    * @type (Event|Object)
 19    */
 20   this.originalEvent = eventObj;
 21   /**
 22    * A boolean indicating whether the event bubbles up through the DOM or not.
 23    * @type boolean
 24    */
 25   this.bubbles = (function () {
 26     if (eventObj.bubbles) {
 27       return eventObj.bubbles;
 28     }
 29     if (eventObj.cancelBubble) {
 30       return eventObj.cancelBubble;
 31     }
 32     return false;
 33   })();
 34   /**
 35    * A reference to the currently registered target for the event.
 36    * @type Element|window|document
 37    */
 38   this.currentTarget = eventObj.currentTarget || eventObj.srcElement;
 39   /**
 40    * Indicates whether or not <code>event.preventDefault()</code> has been
 41    *   called on the event.
 42    * @type boolean
 43    */
 44   this.defaultPrevented = false;
 45   /**
 46    * Indicates which phase of the event flow is being processed.
 47    * @type number
 48    * @see https://developer.mozilla.org/en/DOM/event.eventPhase
 49    */
 50   this.eventPhase = 1; // TODO implement somehow
 51   /**
 52    * A reference to the target to which the event was originally dispatched.
 53    * @type Element|window|document
 54    */
 55   this.target = eventObj.target || eventObj.srcElement;
 56   /**
 57    * The time that the event was created.
 58    * @type number
 59    */
 60   this.timeStamp = (new Date()).getTime();
 61   /**
 62    * The name of the event.
 63    * @type string
 64    */
 65   this.type = eventObj.type.toLowerCase();
 66   /**
 67    * A boolean indicating whether the event is cancelable.
 68    * @type boolean
 69    */
 70   this.cancelable = eventObj.cancelable !== undefined ? eventObj.cancelable : (function (type) {
 71     switch (type) {
 72       // TODO add the rest
 73       case 'click':
 74       case 'submit':
 75         return true;
 76     }
 77 
 78     return false;
 79   })(this.type);
 80   /**
 81    * Indicates whether or not the event was initiated by the browser.
 82    * @type boolean
 83    */
 84   this.isTrusted = false;
 85 
 86   return this;
 87 };
 88 /**
 89  * Cancels the event (if it is cancelable).
 90  */
 91 sEvent.prototype.preventDefault = function () {
 92   this.defaultPrevented = true;
 93 
 94   if (this._isIE) {
 95     this.originalEvent.returnValue = false;
 96     return;
 97   }
 98 
 99   if (this.originalEvent.preventDefault) {
100     this.originalEvent.preventDefault();
101   }
102 };
103 /**
104  * Prevents other listeners of the same event to be called. Note that in IE
105  *   prior to version 9, this does the same thing as stopPropagation().
106  */
107 sEvent.prototype.stopImmediatePropagation = function () {
108   if (this._isIE) {
109     this.originalEvent.cancelBubble = true;
110   }
111 
112   if (this.originalEvent.stopImmediatePropagation) {
113     this.originalEvent.stopImmediatePropagation();
114   }
115 };
116 /**
117  * Prevents further propagation of the current event. Note that in IE prior to
118  *   version 9, this does the same thing as stopImmediatePropagation().
119  */
120 sEvent.prototype.stopPropagation = function () {
121   if (this._isIE) {
122     this.stopImmediatePropagation();
123   }
124 
125   if (this.originalEvent.stopPropagation) {
126     this.originalEvent.stopPropagation();
127   }
128 };
129 /**
130  * Initialise an event. Does nothing in IE versions prior to 9.
131  * @param {string} type Type of event.
132  * @param {boolean} bubbles A boolean indicating whether the event bubbles up
133  *   through the DOM or not.
134  * @param {boolean} cancelable A boolean indicating whether the event can be
135  *   canceled.
136  */
137 sEvent.prototype.initEvent = function (type, bubbles, cancelable) {
138   if (this._isIE) {
139     return;
140   }
141 
142   if (this.originalEvent.initEvent) {
143     this.originalEvent.initEvent(type, bubbles, cancelable);
144   }
145 };
