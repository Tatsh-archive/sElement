1 /**
  2  * HTMLCollection emulation object.
  3  * @constructor
  4  * @param {Element} element DOM element.
  5  * @param {Array} items Array of elements to be managed.
  6  */
  7 var sHTMLCollection = function (element, items) {
  8   /**
  9    * @type Array
 10    * @private
 11    */
 12   this._items = items;
 13   /**
 14    * Number of elements.
 15    * @type number
 16    */
 17   this.length = items.length;
 18   /**
 19    * @type Element
 20    * @private
 21    */
 22   this._element = element;
 23 };
 24 /**
 25  * Gets the correct object depending on the environment.
 26  * @param {Element} element DOM element.
 27  * @param {string} member Member to check for.
 28  * @param {Array|NodeList} fallback Array or iterable object that contains
 29  *   elements to manage.
 30  * @returns {sHTMLCollection|HTMLCollection} The native browser object or the
 31  *   sHTMLCollection emulation class.
 32  */
 33 sHTMLCollection.getCorrectObject = function (element, member, fallback) {
 34   var find = [
 35     'item',
 36     'namedItem',
 37     'length'
 38   ];
 39   var items = [];
 40   var hasSupport = true;
 41   var i = 0;
 42 
 43   if (element[member]) {
 44     for (i = 0; i < find.length; i++) {
 45       if (!element[member][find[i]]) {
 46         hasSupport = false;
 47       }
 48     }
 49   }
 50   else {
 51     hasSupport = false;
 52   }
 53 
 54   if (hasSupport) {
 55     return element[member];
 56   }
 57   else {
 58     for (i = 0; i < fallback.length; i++) {
 59       items.push(fallback[i]);
 60     }
 61   }
 62 
 63   return new sHTMLCollection(element, items);
 64 };
 65 /**
 66  * Returns the specific node at the given zero-based <code>index</code> into
 67  *   the list.
 68  * @param {number} index Index.
 69  * @returns {Element|null} Null if the index is out of range, element
 70  *   otherwise.
 71  */
 72 sHTMLCollection.prototype.item = function (index) {
 73   if (!this._items[index]) {
 74     return null;
 75   }
 76   return this._items[index];
 77 };
 78 /**
 79  * Returns the specific node whose ID or, as a fallback, name matches the
 80  *   string specified by <code>name</code>. Matching by name is only done as a
 81  *   last resort, only in HTML, and only if the referenced element supports the
 82  *   <code>name</code> attribute.
 83  * @param {string} name ID or name to find.
 84  * @returns {Element|null} Element if it is found, null if no such node exists.
 85  */
 86 sHTMLCollection.prototype.namedItem = function (name) {
 87   var i;
 88 
 89   for (i = 0; i < this.length; i++) {
 90     if (this._items[i].getAttribute('id') === name) {
 91       return this._items[i];
 92     }
 93   }
 94 
 95   for (i = 0; i < this.length; i++) {
 96     if (this._items[i].getAttribute('name') === name) {
 97       return this._items[i];
 98     }
 99   }
100 
101   return null;
102 };
