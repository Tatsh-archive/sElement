1 /**
  2  * Class list to emulate DOMTokenList returned by classList for elements.
  3  * @constructor
  4  * @param {Element} [element=null] DOM element.
  5  * @returns {DOMTokenList|sDOMTokenList} The native browser DOMTokenList or
  6  *   sDOMTokenList.
  7  */
  8 var sDOMTokenList = function (element) {
  9   if (!element) {
 10     return this;
 11   }
 12 
 13   /**
 14    * Whether or not this is using a native host object. Only needed for
 15    *   subclasses.
 16    * @type boolean
 17    * @private
 18    */
 19   this._isNative = false;
 20 
 21   if (element.classList) {
 22     this._isNative = true;
 23     return element.classList;
 24   }
 25 
 26   /**
 27    * The element being managed.
 28    * @type Element
 29    * @private
 30    */
 31   this._DOMElement = element;
 32 
 33   /**
 34    * Array of class names.
 35    * @type Array
 36    * @private
 37    */
 38   this._classes = element.className.split(/\s+/);
 39 
 40   /**
 41    * Number of classes.
 42    * @type number
 43    */
 44   this.length = this._classes.length;
 45 
 46   // Clean up
 47   /**
 48    * @type Array
 49    * @private
 50    */
 51   var fixed = [];
 52   for (var i = 0; i < this.length; i++) {
 53     if (!this._classes[i]) {
 54       continue;
 55     }
 56     fixed.push(fUTF8.trim(this._classes[i]));
 57   }
 58   this._classes = fixed;
 59 
 60   return this;
 61 };
 62 /**
 63  * Returns an item in the list by its index.
 64  * @param {number} idx Index.
 65  * @returns {string|undefined} Item requested or undefined if the number is
 66  *   greater than or equal to length.
 67  */
 68 sDOMTokenList.prototype.item = function (idx) {
 69   if (idx >= this.length) {
 70     return undefined;
 71   }
 72   return this._classes[idx];
 73 };
 74 /**
 75  * Check if a token exists.
 76  * @param {string} token Token to check for.
 77  * @returns {boolean} True or false.
 78  */
 79 sDOMTokenList.prototype.contains = function (token) {
 80   for (var i = 0; i < this.length; i++) {
 81     if (this._classes[i] === token) {
 82       return true;
 83     }
 84   }
 85   return false;
 86 };
